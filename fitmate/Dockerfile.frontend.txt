# ============================================
FROM node:20-alpine AS build

# Set working directory
WORKDIR /app

# Copy dependency files first for better caching
COPY package*.json ./

# Install dependencies
RUN npm ci --legacy-peer-deps

# Copy the remaining source code
COPY . .

# Build the Angular app for production
RUN npm run build -- --configuration production

# ============================================
# üåê Stage 2: Serve the app with NGINX
# ============================================
FROM nginx:alpine

# Remove default NGINX page
RUN rm -rf /usr/share/nginx/html/*

# Copy Angular build output to NGINX html directory
# ‚ö†Ô∏è Update "frontend" to your Angular app name inside "dist" folder
COPY --from=build /app/dist/fitmate/browser /usr/share/nginx/html

# Copy custom NGINX configuration (optional but recommended)
# COPY nginx.conf /etc/nginx/conf.d/default.conf

# Expose HTTP port
EXPOSE 80

# Start NGINX
CMD ["nginx", "-g", "daemon off;"]
